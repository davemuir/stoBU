This sample is to set the line color and the fill colors of itemRenderer.
lineStroke: sets the line color.
fills: sets the fill colors of itemRenderer.
stroke: sets the line color of itemRenderer.

<Line2DSeries yField="Profit" itemRenderer="CircleItemRenderer">
	<lineStroke>
		<Stroke color="0xFF5050" weight="2"/>
	</lineStroke>
	<fills>
		<SolidColor color="0x316a9d"/>
		<SolidColor color="0xffca9b"/>
		<SolidColor color="0xff66cc"/>
	</fills>
	<stroke>
		<Stroke color="0xFF5050" weight="2"/>
	</stroke>
</Line2DSeries>
