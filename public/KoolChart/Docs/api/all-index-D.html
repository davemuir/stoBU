     <Column2DChart id="sub">
      <horizontalAxis>
       <CategoryAxis categoryField="Date" id="naviHAxis"  padding="0"/> 
      </horizontalAxis>
      <verticalAxis>
       <LinearAxis id="vAxis"/>
      </verticalAxis>
      <series>
       <Candle2DSeries name="A" yField="Data1"/>
      </series>
     </Column2DChart>
    </subChart>
    <dataSelector>
     <DualScrollBar visibleItemSize="30"/>

   </DualChart>
